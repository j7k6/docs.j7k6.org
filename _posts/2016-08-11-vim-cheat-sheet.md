```
# fix mixed indentation
gg=G

# set indentation to use 2 spaces
:set tabstop=2 shiftwidth=2 expandtab

# comment multiple lines
1. [ctrl+v]
2. select lines with cursor
3. [shift+i]
4. [#]
5. [esx]

# search & replace
:%s/<$OLD>/<$NEW>/g

# sort lines alphabetically
1. [shift+v]
2. :sort
```
